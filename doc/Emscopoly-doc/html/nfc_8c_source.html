252uint8_t readNFC() { // Beginne Loop-Funktion
253 uint8_t success; // Variable anlegen
254 uint8_t uid[8];
255 uint8_t uidLength;
256
257 uint8_t complete;
258
259 while (!complete) {
260 success = nfc.readPassiveTargetID(PN532_MIFARE_ISO14443A, uid, &uidLength);
261
262
263 if (success) {
264 if (uidLength == 7) {
265
266 uint8_t data[32];
267
268 nfc.ntag2xx_ReadPage(6, data); // Read block 6 - the first user writable block
269
270 uint8_t first = (char)data[0] - '0'; // Converts the first 2 HEX bits to char and subtracts '0' to get the int value
271 uint8_t second = (char)data[1] - '0';
272
273 uint8_t complete = (first * 10) + second; //concats two ints
274
275 for (uint8_t i = 0; i < sizeof(validIDs); i++) {
276 Serial.println(String(i)); // !!do not remove, program doesn't work without it for some reason!!
277 if (validIDs[i] == complete) return complete;
278 }
279
280 complete = 0;
281 }
282 }
283 }
284
285 return complete;
286}

313/*
314 Function: transaction
315
316 Subtracts amount from sender and adds amount to receiver
317
318 sender: id from player who wants to send money
319 receiver: id from player who receives the money
320 amount: how much money to send
321
322 returns: if transaction was successful
323
324*/
325
326bool transaction(uint8_t sender, uint8_t receiver, uint32_t amount) {
327 uint32_t senderCredits = bank[sender];
328 uint32_t receiverCredits = bank[receiver];
329
330 if (senderCredits < amount) return false;
331
332 bank[sender] = bank[sender] - amount;
333 bank[receiver] = bank[receiver] + amount;
334
335 return true;
336}
